cmake_minimum_required(VERSION 3.16)

project(GameLauncher VERSION 1.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)


set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)

set(CMAKE_PREFIX_PATH "C:/Qt/6.10.2/msvc2022_64")

set(Qt6_DIR "C:/Qt/6.10.2/msvc2022_64/lib/cmake/Qt6")

# Qt
find_package(Qt6 REQUIRED COMPONENTS Widgets)
qt_standard_project_setup()

# Sources
set(SOURCES
    src/main.cpp
    src/app/application.cpp
    src/domain/game.cpp
    src/domain/gameLibrary.cpp
    src/services/gameLauncherService.cpp
    src/services/jsonGameRepository.cpp
    src/ui/gameListWidget.cpp
    src/ui/mainwindow.cpp
)

add_executable(${PROJECT_NAME} ${SOURCES})


# Include directories
target_include_directories(${PROJECT_NAME} PRIVATE
    ${CMAKE_SOURCE_DIR}/src
    ${CMAKE_SOURCE_DIR}/external
)

# Link Qt
target_link_libraries(${PROJECT_NAME} PRIVATE Qt6::Widgets)

if(WIN32)
    qt_finalize_executable(${PROJECT_NAME})
endif()

# Output directory
set_target_properties(${PROJECT_NAME} PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin"
)

# Copy config folder next to executable
add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
    ${CMAKE_SOURCE_DIR}/config/game.json
    $<TARGET_FILE_DIR:${PROJECT_NAME}>/config/game.json
)
